	car=""
	Escribir "Ingrese una vocal";
	Leer car;
	Segun car Hacer
		"A":
			Escribir "el valor Ascii de A es 65 ";

			Escribir "el valor Ascii de E es 69 ";
		"I":

		"U":
			Escribir "el valor Ascii de U es 85 ";
		"a":
			Escribir "el valor Ascii de a es 97 ";
		"e":

FinProceso

		FinSi
